```typescript
async function saveGameHistory(data: {
  roomCode: string
  hostId: string
  categoryId: string | null
  categoryName: string | null
  playerCount: number
  winnerId: string | null
  leaderboard: Array<{ id: string; name: string; emoji: string; score: number; rank: number }>
}) {
  const session = await prisma.gameSession.create({
    data: {
      roomCode: data.roomCode,
      hostId: data.hostId,
      categoryId: data.categoryId,
      categoryName: data.categoryName,
      playerCount: data.playerCount,
      winnerId: data.winnerId,
    },
  })

  // Only save results for players — skip if no leaderboard entries
  if (data.leaderboard.length > 0) {
    await prisma.playerGameResult.createMany({
      data: data.leaderboard.map(entry => ({
        gameSessionId: session.id,
        userId: entry.id, // reconnectToken/playerId — may not be a real User.id
        playerName: entry.name,
        playerEmoji: entry.emoji,
        score: entry.score,
        rank: entry.rank,
        isWinner: entry.rank === 1,
      })),
    })
  }
}
```

**5. Add game:reset handler** (add after the game:end handler):

```typescript
// game:reset — Host resets the room for another game
socket.on('game:reset', async () => {
  if (!requireHost(socket)) return

  const roomCode: string = socket.data.roomCode
  if (!roomCode) return

  try {
    const room = await getRoom(roomCode)
    if (!room) {
      socket.emit('room:error', { message: 'Room not found' })
      return
    }

    // Clear game state from Redis
    await saveGameState(roomCode, null as any) // or use a deleteGameState if available
    await updateRoomStatus(roomCode, 'waiting')

    // Clean up module-level state
    clearAutoRevealTimer(roomCode)
    clearVotingTimer(roomCode)
    questionCache.delete(roomCode)
    previousRankings.delete(roomCode)

    // Notify all clients to return to lobby
    io.to(roomCode).emit('room:reset')
    console.log(`[INFO] Game: room reset for play again in ${roomCode}`)
  } catch (err) {
    const message = err instanceof Error ? err.message : 'Failed to reset room'
    socket.emit('room:error', { message })
  }
})
```

Check if `saveGameState` accepts null or if there's a `deleteGameState` function. If neither works, use `redis.del()` directly with the game state key pattern.
