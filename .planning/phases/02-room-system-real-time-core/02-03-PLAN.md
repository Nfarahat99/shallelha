Socket.io Event Contracts (CREATED BY PLAN 02):

Client -> Server events:
- `room:create` — payload: `{}` (hostId from socket.data.userId) — response: `room:created { roomCode }`
- `room:join` — payload: `{ roomCode, name, emoji }` — response: `room:joined { reconnectToken, players }`
- `room:leave` — payload: `{ roomCode }`
- `room:reconnect` — payload: `{ roomCode, reconnectToken }` — response: `room:reconnected { players }`
- `game:start` — payload: `{ roomCode }` — response: `game:started { roomCode }` (broadcast)
- `game:pause` — payload: `{ roomCode }` — response: `game:paused { roomCode }` (broadcast)
- `game:end` — payload: `{ roomCode }` — response: `game:ended { roomCode }` (broadcast)

Server -> Client events:
- `room:created` — `{ roomCode }` — to creating host only
- `room:joined` — `{ reconnectToken, players }` — to joining player only
- `room:reconnected` — `{ players }` — to reconnecting player only
- `lobby:update` — `{ players }` — broadcast to all room members
- `game:started` — `{ roomCode }` — broadcast to all room members
- `game:paused` — `{ roomCode }` — broadcast to all room members
- `game:ended` — `{ roomCode }` — broadcast to all room members
- `room:error` — `{ message }` — to originating socket only

**Step 1: Update `apps/web/lib/socket.ts` to support auth object.**

Per RESEARCH.md Pattern 6 and locked decision D-04, modify the singleton to accept and update userId:

```typescript
'use client'
import { io, type Socket } from 'socket.io-client'

let socket: Socket | null = null

export function getSocket(userId?: string): Socket {
  if (!socket) {
    const backendUrl = process.env.NEXT_PUBLIC_BACKEND_URL
    if (!backendUrl) {
      throw new Error(
        'NEXT_PUBLIC_BACKEND_URL is not set. ' +
        'Add it to .env.local (local dev) or Vercel environment variables (production).'
      )
    }
    socket = io(backendUrl, {
      transports: ['websocket'],
      autoConnect: false,
      auth: userId ? { userId } : undefined,
    })
  }
  return socket
}

export function updateSocketAuth(userId: string) {
  const s = getSocket()
  s.auth = { userId }
}

export function resetSocket() {
  if (socket) {
    socket.disconnect()
    socket = null
  }
}
```

CRITICAL: `auth: { userId }` — this is read by the server as `socket.handshake.auth.userId`. Do NOT use `extraHeaders` — browser WebSocket silently drops custom headers (per RESEARCH.md critical finding #2).

```typescript
    socket.emit('room:create', {})

    socket.on('room:created', ({ roomCode }: { roomCode: string }) => {
      router.push(`/host/${roomCode}`)
    })

    socket.on('room:error', ({ message }: { message: string }) => {
      setError(message)
      setCreating(false)
    })

    // Timeout fallback
    setTimeout(() => {
      if (creating) {
        setError('فشل إنشاء الغرفة — حاول مرة أخرى')
        setCreating(false)
      }
    }, 5000)
  }

  return (
    <div className="flex flex-col items-center gap-4">
      <button
        onClick={handleCreate}
        disabled={creating}
        className="px-8 py-4 bg-amber-500 text-white font-bold text-xl rounded-xl
                   disabled:opacity-50 hover:bg-amber-600 active:scale-95 transition-all"
      >
        {creating ? 'جاري الإنشاء...' : 'أنشئ غرفة'}
      </button>
      {error && <p className="text-red-500 font-semibold">{error}</p>}
    </div>
  )
}
```

Create `apps/web/app/join/[roomCode]/PlayerJoin.tsx` (client component):

```typescript
'use client'
import { useEffect, useState, useRef } from 'react'
import { getSocket } from '@/lib/socket'
import { EmojiPicker } from '@/components/lobby/EmojiPicker'
import { PlayerCard } from '@/components/lobby/PlayerCard'

interface Player {
  socketId: string
  name: string
  emoji: string
  disconnectedAt: string | null
}

const STORAGE_KEY = (code: string) => `shllahaReconnectToken_${code}`

export function PlayerJoin({ roomCode }: { roomCode: string }) {
  const [phase, setPhase] = useState<'form' | 'lobby'>('form')
  const [name, setName] = useState('')
  const [emoji, setEmoji] = useState<string | null>(null)
  const [players, setPlayers] = useState<Player[]>([])
  const [gameStatus, setGameStatus] = useState<'waiting' | 'playing' | 'ended'>('waiting')
  const [error, setError] = useState<string | null>(null)
  const [joining, setJoining] = useState(false)
  const socketRef = useRef(getSocket())

  // On mount, check for existing reconnect token
  useEffect(() => {
    const token = sessionStorage.getItem(STORAGE_KEY(roomCode))
    if (token) {
      // Attempt reconnect
      const socket = socketRef.current
      socket.connect()
      socket.emit('room:reconnect', { roomCode, reconnectToken: token })

      socket.on('room:reconnected', ({ players: updated }: { players: Player[] }) => {
        setPlayers(updated)
        setPhase('lobby')
      })

      socket.on('room:error', () => {
        // Reconnect failed — clear token, show form
        sessionStorage.removeItem(STORAGE_KEY(roomCode))
      })
    }

    return () => {
      socketRef.current.off('room:reconnected')
      socketRef.current.off('room:error')
    }
  }, [roomCode])

  // Listen for lobby updates and game events when in lobby
  useEffect(() => {
    if (phase !== 'lobby') return

    const socket = socketRef.current

    socket.on('lobby:update', ({ players: updated }: { players: Player[] }) => {
      setPlayers(updated)
    })

    socket.on('game:started', () => setGameStatus('playing'))
    socket.on('game:ended', () => setGameStatus('ended'))

    // Handle disconnect -> auto-reconnect via sessionStorage
    socket.on('disconnect', () => {
      const token = sessionStorage.getItem(STORAGE_KEY(roomCode))
      if (token) {
        socket.once('connect', () => {
          socket.emit('room:reconnect', { roomCode, reconnectToken: token })
        })
      }
    })

    return () => {
      socket.off('lobby:update')
      socket.off('game:started')
      socket.off('game:ended')
      socket.off('disconnect')
    }
  }, [phase, roomCode])

  const handleJoin = () => {
    if (!name.trim() || !emoji) return
    if (name.trim().length > 15) {
      setError('الاسم طويل جدا (15 حرف كحد أقصى)')
      return
    }

    setJoining(true)
    setError(null)
    const socket = socketRef.current
    socket.connect()

    socket.emit('room:join', { roomCode, name: name.trim(), emoji })

    socket.on('room:joined', ({ reconnectToken, players: updated }: { reconnectToken: string; players: Player[] }) => {
      sessionStorage.setItem(STORAGE_KEY(roomCode), reconnectToken)
      setPlayers(updated)
      setPhase('lobby')
      setJoining(false)
    })

    socket.on('room:error', ({ message }: { message: string }) => {
      setError(message)
      setJoining(false)
    })
  }

  if (phase === 'form') {
    return (
      <main className="min-h-screen flex flex-col items-center justify-center gap-6 p-8">
        <h1 className="text-2xl font-bold">انضم للغرفة {roomCode}</h1>

        <div className="w-full max-w-sm space-y-6">
          {/* Name input — Arabic RTL */}
          <div>
            <label className="block text-sm font-semibold mb-2">اسمك</label>
            <input
              type="text"
              value={name}
              onChange={(e) => setName(e.target.value.slice(0, 15))}
              placeholder="أدخل اسمك"
              dir="rtl"
              className="w-full px-4 py-3 border-2 border-gray-300 rounded-xl text-lg
                         focus:border-amber-500 focus:outline-none"
              autoFocus
            />
          </div>

          {/* Emoji picker */}
          <div>
            <label className="block text-sm font-semibold mb-2">اختر رمزك</label>
            <EmojiPicker selected={emoji} onSelect={setEmoji} />
          </div>

          {/* Join button */}
          <button
            onClick={handleJoin}
            disabled={!name.trim() || !emoji || joining}
            className="w-full px-6 py-4 bg-amber-500 text-white font-bold text-lg rounded-xl
                       disabled:opacity-50 hover:bg-amber-600 active:scale-95 transition-all"
          >
            {joining ? 'جاري الانضمام...' : 'انضم'}
          </button>

          {error && <p className="text-red-500 text-center font-semibold">{error}</p>}
        </div>
      </main>
    )
  }

  // Lobby phase
  return (
    <main className="min-h-screen p-6 max-w-md mx-auto">
      <div className="text-center mb-6">
        <p className="text-gray-500 text-sm">الغرفة</p>
        <h1 className="text-4xl font-black text-amber-600">{roomCode}</h1>
      </div>

      {gameStatus === 'waiting' && (
        <p className="text-center text-gray-500 mb-6">في انتظار بدء اللعبة...</p>
      )}
      {gameStatus === 'playing' && (
        <p className="text-center text-green-600 font-bold mb-6">اللعبة بدأت!</p>
      )}
      {gameStatus === 'ended' && (
        <p className="text-center text-gray-500 mb-6">انتهت اللعبة</p>
      )}

      <div className="grid gap-3">
        {players
          .filter(p => !p.disconnectedAt)
          .map((player) => (
            <PlayerCard
              key={player.socketId}
              name={player.name}
              emoji={player.emoji}
            />
          ))}
      </div>
    </main>
  )
}
```
