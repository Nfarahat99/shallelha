GET /cards/result endpoint:
- Query params: ?gameId=, ?variant=snapchat|whatsapp (default: whatsapp)

  ```typescript
  const shareCard = async () => {
    const url = `${backendUrl}/cards/result?gameId=${gameId}&variant=${variant}`
    const response = await fetch(url)
    const blob = await response.blob()
    const file = new File([blob], `shalelha-result.png`, { type: 'image/png' })
    if (navigator.canShare?.({ files: [file] })) {
      await navigator.share({ files: [file], title: 'شعللها — نتائج اللعبة' })
    } else {
      // Fallback: download
      const a = document.createElement('a')
      a.href = URL.createObjectURL(blob)
      a.download = `shalelha-result-${variant}.png`
      a.click()
    }
  }
  ```
