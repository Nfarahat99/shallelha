```typescript
import { test, expect } from '@playwright/test'

const BASE_URL = process.env.BASE_URL ?? 'http://localhost:3000'

test.describe('/api/leaderboard', () => {
  test('alltime period returns JSON array with required fields', async ({ request }) => {
    const res = await request.get(`${BASE_URL}/api/leaderboard?period=alltime`)
    expect(res.status()).toBe(200)
    const data = await res.json()
    expect(Array.isArray(data)).toBe(true)
    // May be empty if no game data — just verify shape if data present
    if (data.length > 0) {
      const first = data[0]
      expect(typeof first.rank).toBe('number')
      expect(typeof first.name).toBe('string')
      expect(typeof first.winCount).toBe('number')
      expect(typeof first.totalGamesPlayed).toBe('number')
    }
  })

  test('weekly period returns JSON array', async ({ request }) => {
    const res = await request.get(`${BASE_URL}/api/leaderboard?period=weekly`)
    expect(res.status()).toBe(200)
    const data = await res.json()
    expect(Array.isArray(data)).toBe(true)
  })

  test('invalid period returns 400', async ({ request }) => {
    const res = await request.get(`${BASE_URL}/api/leaderboard?period=invalid`)
    expect(res.status()).toBe(400)
  })

  test('category filter does not crash', async ({ request }) => {
    const res = await request.get(`${BASE_URL}/api/leaderboard?period=alltime&category=test`)
    // Either 200 (valid category, may return empty) or 404 (unknown category)
    expect([200, 404]).toContain(res.status())
  })
})
```

```typescript
    expect(response?.status()).not.toBe(500)
    // Page title or heading in Arabic should be present
    await expect(page.locator('h1, h2')).not.toHaveCount(0)
  })

  test('/leaderboard all-time toggle is present', async ({ page }) => {
    await page.goto(`${BASE_URL}/leaderboard`)
    // Should find period toggle buttons — text may be "كل الأوقات" or "هذا الأسبوع"
    const buttons = page.locator('button')
    await expect(buttons).not.toHaveCount(0)
  })
})

test.describe('Phase 12 — Profile OG card', () => {
  test('/api/og/profile with missing userId returns 400', async ({ request }) => {
    const res = await request.get(`${BASE_URL}/api/og/profile`)
    expect(res.status()).toBe(400)
  })

  test('/api/og/profile with unknown userId returns 404 or image', async ({ request }) => {
    const res = await request.get(`${BASE_URL}/api/og/profile?userId=nonexistent-user-id-abc123`)
    // Either 404 (user not found) or 200 (image generated with defaults)
    expect([200, 404]).toContain(res.status())
  })
})

test.describe('Phase 12 — Avatar null safety', () => {
  test('/leaderboard does not crash with null avatar data', async ({ page }) => {
    // This test validates that components handle null avatarConfig gracefully
    const response = await page.goto(`${BASE_URL}/leaderboard`)
    // No uncaught errors
    const errors: string[] = []
    page.on('pageerror', (err) => errors.push(err.message))
    await page.waitForLoadState('networkidle')
    expect(errors.filter(e => e.includes('Cannot read'))).toHaveLength(0)
  })
})
```
