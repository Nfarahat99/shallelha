```typescript
export async function claimAnonymousStats(data: {
  gameSessionId: string
  playerName: string
}): Promise<{ claimed: number; error?: string }> {
  const session = await auth()
  if (!session?.user?.id) return { claimed: 0, error: 'Unauthorized' }

  // Sanitize inputs
  const sessionId = data.gameSessionId.replace(/[^a-zA-Z0-9_-]/g, '').slice(0, 40)
  const playerName = data.playerName.trim().slice(0, 50)
  if (!sessionId || !playerName) return { claimed: 0, error: 'Invalid input' }

  const sevenDaysAgo = new Date(Date.now() - 7 * 24 * 60 * 60 * 1000)

  // Find unclaimed results matching this session + player name
  const candidates = await prisma.playerGameResult.findMany({
    where: {
      gameSessionId: sessionId,
      userId: null,
      playerName: { equals: playerName, mode: 'insensitive' },
      createdAt: { gte: sevenDaysAgo },
    },
  })

  if (candidates.length === 0) return { claimed: 0 }

  // Claim them
  const result = await prisma.playerGameResult.updateMany({
    where: {
      id: { in: candidates.map((r) => r.id) },
      userId: null, // double-check — prevent race condition
    },
    data: { userId: session.user.id },
  })

  if (result.count === 0) return { claimed: 0 }

  // Recalculate aggregate stats from DB to avoid drift
  const [statsResult] = await prisma.$queryRaw<[{ games: bigint; wins: bigint }]>`
    SELECT COUNT(*) AS games, COUNT(*) FILTER (WHERE "isWinner" = true) AS wins
    FROM "PlayerGameResult"
    WHERE "userId" = ${session.user.id}
  `
  const bestStreakResult = await prisma.playerGameResult.aggregate({
    where: { userId: session.user.id },
    _max: { score: true }, // score proxy — bestStreak is on User, keep existing value
  })

  await prisma.user.update({
    where: { id: session.user.id },
    data: {
      totalGamesPlayed: Number(statsResult.games),
      winCount: Number(statsResult.wins),
    },
  })

  revalidatePath('/profile')
  return { claimed: result.count }
}
```

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-12-06-01 | Spoofing | Claiming another player's stats by guessing playerName | mitigate | Claim is scoped to gameSessionId (which the claimant must have participated in); 7-day window limits exposure; playerName must match exactly (case-insensitive) within the same session |
| T-12-06-02 | Tampering | Race condition — two users claim same result simultaneously | mitigate | updateMany WHERE userId IS NULL acts as an optimistic lock; second claim updates 0 rows and returns claimed: 0 |
| T-12-06-03 | Tampering | SQL injection via playerName/gameSessionId | mitigate | gameSessionId sanitized to alphanumeric; playerName trimmed; Prisma ORM parameterizes all queries |
| T-12-06-04 | Information Disclosure | userId=null results visible in leaderboard | accept | Leaderboard filters WHERE userId IS NOT NULL — anonymous rows never appear in public ranking |
