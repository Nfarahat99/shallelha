```tsx
'use client'
import { useState, useEffect } from 'react'
import { socket } from '@/lib/socket'

interface BluffingSubmitScreenProps {
  question: { text: string }
}

export function BluffingSubmitScreen({ question }: BluffingSubmitScreenProps) {
  const [role, setRole] = useState<'truth' | 'bluffer' | null>(null)
  const [realAnswer, setRealAnswer] = useState<string | null>(null)
  const [instruction, setInstruction] = useState<string>('اكتب إجابة مقنعة...')
  const [text, setText] = useState('')
  const [submitted, setSubmitted] = useState(false)
  const [submittedCount, setSubmittedCount] = useState(0)
  const [totalPlayers, setTotalPlayers] = useState(0)

  useEffect(() => {
    socket.on('bluffing:role', (data: { role: 'truth' | 'bluffer'; answer?: string; instruction: string }) => {
      setRole(data.role)
      if (data.answer) setRealAnswer(data.answer)
      setInstruction(data.instruction)
    })
    socket.on('bluffing:submitted', () => setSubmitted(true))
    socket.on('bluffing:progress', ({ submitted, total }: { submitted: number; total: number }) => {
      setSubmittedCount(submitted)
      setTotalPlayers(total)
    })
    return () => {
      socket.off('bluffing:role')
      socket.off('bluffing:submitted')
      socket.off('bluffing:progress')
    }
  }, [])

  function handleSubmit(e: React.FormEvent) {
    e.preventDefault()
    if (!text.trim() || submitted) return
    socket.emit('bluffing:submit', { text: text.trim() })
  }

  return (
    <div dir="rtl" className="flex flex-col gap-4 p-4">
      {/* Question */}
      <div className="bg-white/5 border border-white/10 rounded-2xl p-4 text-center">
        <p className="text-white/50 text-xs mb-1">السؤال</p>
        <p className="text-white text-lg font-bold">{question.text}</p>
      </div>

      {/* Role badge */}
      {role === 'truth' && (
        <div className="bg-green-500/20 border border-green-400/30 rounded-xl p-3 text-center">
          <p className="text-green-300 text-sm font-bold">أنت تعرف الإجابة الحقيقية</p>
          <p className="text-green-400 text-xl font-bold mt-1">{realAnswer}</p>
          <p className="text-green-200/70 text-xs mt-1">{instruction}</p>
        </div>
      )}
      {role === 'bluffer' && (
        <div className="bg-yellow-500/20 border border-yellow-400/30 rounded-xl p-3 text-center">
          <p className="text-yellow-300 text-sm font-bold">أنت كاذب!</p>
          <p className="text-yellow-200/70 text-xs mt-1">{instruction}</p>
        </div>
      )}

      {/* Answer input */}
      {!submitted ? (
        <form onSubmit={handleSubmit} className="flex flex-col gap-3">
          <textarea
            value={text}
            onChange={(e) => setText(e.target.value)}
            placeholder="اكتب إجابتك هنا..."
            maxLength={200}
            rows={3}
            className="w-full bg-white/10 border border-white/20 rounded-xl px-4 py-3 text-white placeholder-white/40 text-right outline-none focus:ring-2 focus:ring-brand-400 resize-none"
          />
          <button
            type="submit"
            disabled={!text.trim()}
            className="bg-brand-500 text-white py-3 rounded-xl font-bold disabled:opacity-50 hover:bg-brand-400 active:scale-95 transition-all"
          >
            إرسال الإجابة
          </button>
        </form>
      ) : (
        <div className="bg-white/5 border border-white/10 rounded-xl p-4 text-center">
          <p className="text-white/60 text-sm">تم إرسال إجابتك</p>
          <p className="text-white/40 text-xs mt-1">
            {submittedCount} / {totalPlayers} أجابوا
          </p>
        </div>
      )}

      {/* Progress */}
      {submitted && totalPlayers > 0 && (
        <div className="bg-white/5 rounded-full overflow-hidden h-2">
          <div
            className="bg-brand-400 h-2 transition-all duration-500"
            style={{ width: `${(submittedCount / totalPlayers) * 100}%` }}
          />
        </div>
      )}
    </div>
  )
}
```

```tsx
'use client'
import { useState, useEffect } from 'react'
import { socket } from '@/lib/socket'

interface Answer { id: string; text: string }
interface BluffingResult {
  playerId: string; playerName: string; playerEmoji: string
  text: string; votes: string[]; isReal: boolean
}

export function BluffingVoteScreen() {
  const [answers, setAnswers] = useState<Answer[]>([])
  const [votedFor, setVotedFor] = useState<string | null>(null)
  const [deadline, setDeadline] = useState<number | null>(null)
  const [results, setResults] = useState<BluffingResult[] | null>(null)
  const [majorityFoundTruth, setMajorityFoundTruth] = useState(false)
  const [votedCount, setVotedCount] = useState(0)
  const [totalVoters, setTotalVoters] = useState(0)
  const [timeLeft, setTimeLeft] = useState(20)

  useEffect(() => {
    socket.on('bluffing:vote_start', (data: { answers: Answer[]; deadline: number }) => {
      setAnswers(data.answers)
      setDeadline(data.deadline)
    })
    socket.on('bluffing:vote_progress', ({ voted, total }: { voted: number; total: number }) => {
      setVotedCount(voted); setTotalVoters(total)
    })
    socket.on('bluffing:results', (data: { submissions: BluffingResult[]; realAnswerHolderId: string; majorityFoundTruth: boolean }) => {
      setResults(data.submissions)
      setMajorityFoundTruth(data.majorityFoundTruth)
    })
    return () => {
      socket.off('bluffing:vote_start')
      socket.off('bluffing:vote_progress')
      socket.off('bluffing:results')
    }
  }, [])

  // Countdown timer
  useEffect(() => {
    if (!deadline) return
    const interval = setInterval(() => {
      const remaining = Math.max(0, Math.ceil((deadline - Date.now()) / 1000))
      setTimeLeft(remaining)
      if (remaining === 0) clearInterval(interval)
    }, 500)
    return () => clearInterval(interval)
  }, [deadline])

  function handleVote(targetId: string) {
    if (votedFor) return
    socket.emit('bluffing:vote', { targetPlayerId: targetId })
    setVotedFor(targetId)
  }

  if (results) {
    return (
      <div dir="rtl" className="flex flex-col gap-3 p-4">
        <div className={`rounded-xl p-3 text-center border ${majorityFoundTruth ? 'bg-green-500/20 border-green-400/30' : 'bg-yellow-500/20 border-yellow-400/30'}`}>
          <p className={`font-bold text-sm ${majorityFoundTruth ? 'text-green-300' : 'text-yellow-300'}`}>
            {majorityFoundTruth ? 'الأغلبية وجدت الإجابة الحقيقية!' : 'الكاذبون فازوا هذه الجولة!'}
          </p>
        </div>
        {results.map((r) => (
          <div key={r.playerId} className={`rounded-xl p-3 border ${r.isReal ? 'bg-green-500/10 border-green-400/30' : 'bg-white/5 border-white/10'}`}>
            <div className="flex items-center gap-2 mb-1">
              <span>{r.playerEmoji}</span>
              <span className="text-white/70 text-sm">{r.playerName}</span>
              {r.isReal && <span className="text-green-400 text-xs font-bold">الإجابة الحقيقية</span>}
            </div>
            <p className="text-white font-medium">{r.text}</p>
            <p className="text-white/40 text-xs mt-1">{r.votes.length} صوت</p>
          </div>
        ))}
      </div>
    )
  }

  return (
    <div dir="rtl" className="flex flex-col gap-3 p-4">
      <div className="flex justify-between items-center">
        <p className="text-white/60 text-sm">أي إجابة هي الحقيقية؟</p>
        <div className="bg-white/10 rounded-full px-3 py-1">
          <span className="text-white font-mono text-sm">{timeLeft}s</span>
        </div>
      </div>
      {answers.map((answer) => (
        <button
          key={answer.id}
          onClick={() => handleVote(answer.id)}
          disabled={!!votedFor}
          className={`w-full text-right p-4 rounded-xl border transition-all font-medium ${
            votedFor === answer.id
              ? 'bg-brand-500/30 border-brand-400 text-white'
              : votedFor
              ? 'bg-white/5 border-white/10 text-white/40'
              : 'bg-white/10 border-white/20 text-white hover:bg-white/15 active:scale-[0.98]'
          }`}
        >
          {answer.text}
        </button>
      ))}
      {votedFor && totalVoters > 0 && (
        <p className="text-white/40 text-xs text-center">{votedCount} / {totalVoters} صوّتوا</p>
      )}
    </div>
  )
}
```

```tsx
'use client'
import { useState, useEffect } from 'react'
import { socket } from '@/lib/socket'

interface BluffingResult {
  playerId: string; playerName: string; playerEmoji: string
  text: string; votes: string[]; isReal: boolean
}

interface HostBluffingViewProps {
  question: { text: string }
}

export function HostBluffingView({ question }: HostBluffingViewProps) {
  const [phase, setPhase] = useState<'submit' | 'vote' | 'results'>('submit')
  const [submitted, setSubmitted] = useState(0)
  const [total, setTotal] = useState(0)
  const [voted, setVoted] = useState(0)
  const [results, setResults] = useState<BluffingResult[] | null>(null)
  const [majorityFoundTruth, setMajorityFoundTruth] = useState(false)

  useEffect(() => {
    socket.on('bluffing:progress', ({ submitted: s, total: t }: { submitted: number; total: number }) => {
      setSubmitted(s); setTotal(t)
    })
    socket.on('bluffing:vote_start', () => setPhase('vote'))
    socket.on('bluffing:vote_progress', ({ voted: v }: { voted: number }) => setVoted(v))
    socket.on('bluffing:results', (data: { submissions: BluffingResult[]; majorityFoundTruth: boolean }) => {
      setResults(data.submissions)
      setMajorityFoundTruth(data.majorityFoundTruth)
      setPhase('results')
    })
    return () => {
      socket.off('bluffing:progress')
      socket.off('bluffing:vote_start')
      socket.off('bluffing:vote_progress')
      socket.off('bluffing:results')
    }
  }, [])

  function handleLock() {
    socket.emit('bluffing:lock')
  }

  return (
    <div dir="rtl" className="flex flex-col gap-4 p-6">
      {/* Question */}
      <div className="bg-white/5 border border-white/10 rounded-2xl p-4 text-center">
        <p className="text-white text-xl font-bold">{question.text}</p>
      </div>

      {phase === 'submit' && (
        <div className="flex flex-col gap-4">
          <div className="flex justify-between items-center">
            <span className="text-white/60 text-sm">الإجابات المرسلة</span>
            <span className="text-white font-bold">{submitted} / {total}</span>
          </div>
          {total > 0 && (
            <div className="bg-white/10 rounded-full overflow-hidden h-3">
              <div
                className="bg-brand-400 h-3 transition-all duration-500"
                style={{ width: `${(submitted / total) * 100}%` }}
              />
            </div>
          )}
          <button
            onClick={handleLock}
            className="bg-brand-500 text-white py-3 rounded-xl font-bold hover:bg-brand-400 transition-colors mt-2"
          >
            قفل الإجابات والانتقال للتصويت
          </button>
        </div>
      )}

      {phase === 'vote' && (
        <div className="flex flex-col gap-3">
          <div className="bg-white/5 border border-white/10 rounded-xl p-4 text-center">
            <p className="text-white/60 text-sm">مرحلة التصويت</p>
            <p className="text-white text-lg font-bold mt-1">{voted} صوّتوا</p>
          </div>
        </div>
      )}

      {phase === 'results' && results && (
        <div className="flex flex-col gap-3">
          <div className={`rounded-xl p-3 text-center border ${majorityFoundTruth ? 'bg-green-500/20 border-green-400/30' : 'bg-yellow-500/20 border-yellow-400/30'}`}>
            <p className={`font-bold ${majorityFoundTruth ? 'text-green-300' : 'text-yellow-300'}`}>
              {majorityFoundTruth ? 'الأغلبية وجدت الإجابة الحقيقية!' : 'الكاذبون فازوا!'}
            </p>
          </div>
          {results.sort((a, b) => b.votes.length - a.votes.length).map((r) => (
            <div key={r.playerId} className={`rounded-xl p-3 border flex justify-between items-center ${r.isReal ? 'bg-green-500/10 border-green-400/30' : 'bg-white/5 border-white/10'}`}>
              <div>
                <p className="text-white font-medium">{r.text}</p>
                <p className="text-white/50 text-xs">{r.playerEmoji} {r.playerName} {r.isReal && '✓ الإجابة الحقيقية'}</p>
              </div>
              <div className="bg-white/10 rounded-lg px-3 py-1 text-white font-bold text-sm">
                {r.votes.length} صوت
              </div>
            </div>
          ))}
        </div>
      )}
    </div>
  )
}
```
