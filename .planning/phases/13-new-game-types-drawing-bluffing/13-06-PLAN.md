```typescript
// If room has a gameMix config AND no packId, build a mixed question list
if (room.gameMix && !room.packId) {
  const { trivia, drawing, bluffing } = room.gameMix
  const mixedQuestions: QuestionData[] = []

  // Load trivia questions (MULTIPLE_CHOICE + FREE_TEXT)
  if (trivia > 0) {
    const triviaQs = await prisma.question.findMany({
      where: { status: QuestionStatus.approved, ...(data?.categoryId ? { categoryId: data.categoryId } : {}) },
      select: { id: true, text: true, options: true, correctIndex: true, timerDuration: true, type: true, mediaUrl: true },
    })
    const shuffled = shuffle(triviaQs.map((q) => ({
      id: q.id,
      text: q.text,
      options: q.options as string[],
      correctIndex: q.correctIndex,
      timerDuration: q.timerDuration,
      type: q.type as QuestionData['type'],
      mediaUrl: q.mediaUrl ?? undefined,
    }))).slice(0, trivia)
    mixedQuestions.push(...shuffled)
  }

  // Load drawing prompts
  if (drawing > 0) {
    const drawingPrompts = await prisma.drawingPrompt.findMany({
      where: { archived: false },
      orderBy: { timesUsed: 'asc' },
      take: drawing * 3,
    })
    const shuffledDrawing = shuffle(drawingPrompts).slice(0, drawing)
    for (const prompt of shuffledDrawing) {
      mixedQuestions.push({
        id: prompt.id,
        text: prompt.text,
        options: [],
        correctIndex: -1,
        timerDuration: 60,
        type: 'DRAWING',
      })
      // Increment timesUsed fire-and-forget
      void prisma.drawingPrompt.update({ where: { id: prompt.id }, data: { timesUsed: { increment: 1 } } }).catch(() => {})
    }
  }

  // Load bluffing questions
  if (bluffing > 0) {
    const bluffingQs = await prisma.question.findMany({
      where: { status: QuestionStatus.approved, type: 'BLUFFING' },
      select: { id: true, text: true, options: true, correctIndex: true, timerDuration: true, type: true, mediaUrl: true },
      take: bluffing * 3,
    })
    const shuffledBluffing = shuffle(bluffingQs.map((q) => ({
      id: q.id,
      text: q.text,
      options: q.options as string[],
      correctIndex: q.correctIndex,
      timerDuration: q.timerDuration,
      type: 'BLUFFING' as QuestionData['type'],
      mediaUrl: undefined,
    }))).slice(0, bluffing)
    mixedQuestions.push(...shuffledBluffing)
  }

  if (mixedQuestions.length === 0) {
    socket.emit('game:error', { message: 'لا توجد أسئلة كافية لبدء اللعبة' })
    return
  }

  questions = shuffle(mixedQuestions)
}
```

export function setupSocketHandlers(io: Server): void {
  io.on('connection', (socket) => {
    console.log(`[INFO] Socket connected: ${socket.id}`)

    registerRoomHandlers(io, socket)
    registerGameHandlers(io, socket)
    registerDrawingHandlers(io, socket)
    registerBluffingHandlers(io, socket)

      console.log(`[INFO] Socket disconnected: ${socket.id} — ${reason}`)

        clearAutoRevealTimer(socket.data.roomCode)
        clearVotingTimer(socket.data.roomCode)
        clearDrawingTimer(socket.data.roomCode)

    })

async function fetchPackInfo(packId: string): Promise<PackInfo | null> {
  const backendUrl = process.env.NEXT_PUBLIC_BACKEND_URL
  if (!backendUrl) return null
  try {
    const res = await fetch(`${backendUrl}/packs/${encodeURIComponent(packId)}`)
    if (!res.ok) return null
    return res.json() as Promise<PackInfo>
  } catch { return null }
}

  const [packInfo, setPackInfo] = useState<PackInfo | null>(null)

  const [gameMix, setGameMix] = useState<GameMix>({ trivia: 5, drawing: 2, bluffing: 1 })

    if (!packId) return
    fetchPackInfo(packId).then((info) => {
      if (!info) setPackError('تعذر تحميل معلومات الباقة')
      else if (info.status !== 'APPROVED') setPackError('هذه الباقة غير متاحة')

      socket.off('room:created')
      socket.off('room:error')

            className="inline-flex items-center gap-2 rounded-xl bg-brand-600 hover:bg-brand-500 text-white font-semibold px-5 py-2.5 text-sm transition-all duration-150 min-h-[44px]">
            العودة إلى الباقات

  if (step === 'select' && !packId) {
    const total = gameMix.trivia + gameMix.drawing + gameMix.bluffing
    return (
      <main className="min-h-dvh flex flex-col items-center justify-center bg-gradient-to-b from-gray-950 via-brand-950 to-gray-900 px-4" dir="rtl">
        <div className="w-full max-w-sm space-y-6">
