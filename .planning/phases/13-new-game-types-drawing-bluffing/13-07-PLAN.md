**apps/web/app/(admin)/admin/drawing-prompts/actions.ts:**
```typescript
'use server'

import { prisma } from '@/lib/prisma'
import { revalidatePath } from 'next/cache'

export async function createDrawingPrompt(formData: FormData) {
  const text = (formData.get('text') as string)?.trim()
  const category = (formData.get('category') as string)?.trim()
  const difficulty = (formData.get('difficulty') as string) || 'medium'
  const language = (formData.get('language') as string) || 'ar'

  if (!text) throw new Error('نص الكلمة مطلوب')
  if (!category) throw new Error('الفئة مطلوبة')
  if (!['easy', 'medium', 'hard'].includes(difficulty)) throw new Error('الصعوبة غير صالحة')

  await prisma.drawingPrompt.create({
    data: { text, category, difficulty, language, archived: false },
  })
  revalidatePath('/admin/drawing-prompts')
}

export async function archiveDrawingPrompt(id: string) {
  if (!id || typeof id !== 'string') throw new Error('معرّف غير صالح')
  await prisma.drawingPrompt.update({
    where: { id },
    data: { archived: true },
  })
  revalidatePath('/admin/drawing-prompts')
}

export async function restoreDrawingPrompt(id: string) {
  if (!id || typeof id !== 'string') throw new Error('معرّف غير صالح')
  await prisma.drawingPrompt.update({
    where: { id },
    data: { archived: false },
  })
  revalidatePath('/admin/drawing-prompts')
}
```

```typescript
import { prisma } from '@/lib/prisma'
import { DrawingPromptForm } from './DrawingPromptForm'
import { DrawingPromptList } from './DrawingPromptList'

export const dynamic = 'force-dynamic'

export default async function DrawingPromptsPage({
  searchParams,
}: {
  searchParams: { showArchived?: string; category?: string; difficulty?: string }
}) {
  const showArchived = searchParams.showArchived === '1'
  const prompts = await prisma.drawingPrompt.findMany({
    where: {
      ...(showArchived ? {} : { archived: false }),
      ...(searchParams.category ? { category: searchParams.category } : {}),
      ...(searchParams.difficulty ? { difficulty: searchParams.difficulty } : {}),
    },
    orderBy: [{ category: 'asc' }, { createdAt: 'desc' }],
  })

  return (
    <div dir="rtl">
      <div className="flex items-center justify-between mb-6">
        <h1 className="text-2xl font-bold">كلمات الرسم</h1>
        <span className="text-sm text-gray-500">{prompts.length} كلمة</span>
      </div>
      <DrawingPromptForm />
      <div className="mt-8">
        <DrawingPromptList prompts={prompts} showArchived={showArchived} />
      </div>
    </div>
  )
}
```
